The words - predefined tokens are specified between " and ":
Syntactical rules:

PROGRAM = "{" DECLLIST ";" STMT "}"

DECLLIST = DECLARATION | DECLARATION ";" DECLLIST

DECLARATION = "@"IDENTIFIER ":" TYPE

GENERICTYPE = "bool" | "char" | "int"

ARRAYDECL = GENERICTYPE "[" NUMBER "]"

TYPE = GENERICTYPE | ARRAYDECL

STMT = (SIMPLESTMT | STRUCTSTMT) [{STMT}]

SIMPLESTMT = (ASSIGNSTMT | IOSTMT) ";"

ASSIGNSTMT = IDENTIFIER "=" EXPRESSION

EXPRESSION = EXPRESSIONSIGN {EXPRESSIONSIGN | IDENTIFIER | SIGNNUMBER}

EXPRESSIONSIGN = "+" | "-" | "*" | "/"

IOSTMT = (">>" | "<<") IDENTIFIER

STRUCTSTMT = CMPDSTMT | IFSTMT | WHILESTMT

IFSTMT = "IF" CONDITION "THEN" "{" STMT "}" ["ELSE" "{" STMT "}"]

WHILESTMT = "WHILE" CONDITION "{" STMT "}" 

CONDITION = EXPRESSION RELATION EXPRESSION [LOGICALOPERATOR CONDITION]

RELATION = "<" | "<=" | "==" | "<>" | ">=" | ">"

LOGICALOPERATOR = "&&" | "||"
