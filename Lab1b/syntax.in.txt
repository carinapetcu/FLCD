The words - predefined tokens are specified between " and ":

Sintactical rules:

PROGRAM = "{" DECLLIST ";" CMPDSTMT "}"

DECLLIST = DECLARATION | DECLARATION ";" decllist

DECLARATION = "@"IDENTIFIER ":" TYPE

GENERICTYPE = "bool" | "char" | "int" | "float"

ARRAYDECL = GENERICTYPE "[" NUMBER "]"

TYPE = GENERICTYPE | ARRAYDECL

CMPDSTMT = "{" STMTLIST "}"

STMLIST = STMT | STMT ";" STMTLIST

STMT = SIMPLESTMT | STRUCTSTMT

SIMPLESTMT = ASSIGNSTMT | IOSTMT

ASSIGNSTMT = IDENTIFIER "=" EXPRESSION

EXPRESSION = EXPRESSIONSIGN {EXPRESSIONSIGN | IDENTIFIER | SIGNNUMBER | FLOATSIGNNUMBER}

EXPRESSIONSIGN = "+" | "-" | "*" | "/"

IOSTMT = ">>" | "<<" IDENTIFIER

STRUCTSTMT = CMPDSTMT | IFSTMT | WHILESTMT | FORSTMT

IFSTMT = "IF" CONDITION "THEN" STMT ["ELSE" STMT]

WHILESTMT = "WHILE" CONDITION "DO" STMT

FORSTMT = "FOR" [DECLARATION ","] ASSIGNSTMT "," CONDITION "," ASSIGNSTMT STMT

CONDITION = EXPRESSION RELATION EXPRESSION

RELATION = "<" | "<=" | "==" | "<>" | ">=" | ">"
